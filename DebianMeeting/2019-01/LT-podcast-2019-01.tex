\begin{frame}[fragile]
 \frametitle{なにはともあれ収録}
 \begin{itemize}
  \item Skypeで会話しましょう
	\begin{itemize}
	 \item Debianでうまくできてない……
	 \item マイクデバイスをSkypeアプリに握られてしまう
	\end{itemize}
  \item Skypeの音声を収録するのではなく、\\
	各自自分の声だけ録音しておくと吉
	\begin{itemize}
	 \item 各自の声の音量調整ができるしノイズが入らない
	 \item 相手の声は音洩れしないようイヤホンとかで聞きましょう
	       \begin{itemize}
		\item マイクが相手の声を拾うと悲惨です
	       \end{itemize}
	 \item 音合わせの手段を用意しておくと良いです
	\end{itemize}
  \item 油断してると普通に1時間くらいの長さになってしまう
 \end{itemize}
\end{frame}
